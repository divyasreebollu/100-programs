def weeks_met_target(p1, p2, p3, p4):
    weeks = [p1, p2, p3, p4]
    count = 0
    for week in weeks:
        if week >= 10:
            count += 1
    return count

# Input
p1, p2, p3, p4 = map(int, input().split())

# Output
print(weeks_met_target(p1, p2, p3, p4))
